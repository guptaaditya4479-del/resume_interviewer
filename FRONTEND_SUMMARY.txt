━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

API Base: http://localhost:8000

Function 1: startInterview(resumeFile, domain)
  → POST /interview
  → FormData: { resume: File, domain: string }
  → Returns: { session_id, question }

Function 2: submitAnswer(sessionId, answer)
  → POST /answer
  → JSON: { session_id, answer }
  → Returns: { feedback, scores, next_question?, is_complete }

Function 3: getInterviewSummary(sessionId)
  → GET /interview/:sessionId/summary
  → Returns: { average_score, questions[], summary }

   ────────────────────────────────────────
